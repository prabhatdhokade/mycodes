#include <iostream>
#include<string.h>

using namespace std;


int main()
{
   int i,j,sp=5; /*space*/

    for(i=1;i<=6;i++){

        for(j=1;j<=sp;j++){

            cout<<" ";
        }

        for(j=1;j<=i;j++){

            cout<<i<<" ";

        }
        cout<<"\n";
        sp--;
    }

}
