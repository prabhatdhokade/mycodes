#include <iostream>

using namespace std;

int main()
{
    int max1,max2;
    int arr[25],i,n;
    cout<<"Enter the size of array:";
    cin>>n;
    for(i=0;i<n;i++){
        cin>>arr[i];
    }
    if(arr[0]<arr[1])
    {
        max1=arr[1];
        max2=arr[0];
    }
    else
    {
        max2=arr[1];
        max1=arr[0];
    }
    for(i=2;i<n;i++)
        if(arr[i]>max1){
            max2=max1;
            max1=arr[i];
        }
        else if(arr[i]>max2){
            max2=arr[i];
        }
        
    cout<<"largest arry:"<<max1;
    cout<<"\n 2nd largest arry:"<<max2;
}
