#include <iostream>

using namespace std;

int main()
{
    
     int n1,n2,i,flag,temp;
     cout<<"Enter the  two numbers";
     cin>>n1>>n2;
     
    if(n1>n2)
     {
         temp=n1;
         n1=n2;
         n2=temp;
         
     }
     
     cout<<"Prime numbers between the"<<n1<<"&"<<n2<<"are:";
     
     while(n1<n2)
     {
         flag=0;
         
         for(i=2;i<=n1/2;i++)
         {
             if(n1%i==0)
             {
                flag=1;
                break;
                
             }
         }
         
        if(flag==0)
        
            cout<<"\t"<<n1;
            n1++;
        
     }
}
