#include <iostream>

using namespace std;

int main()
{
    int i,j,len=0;
    char str1[12],str2[12];
    cout<<"Enter a string: ";
    cin>>str1;
    
    for(i=0;str1[i]!='\0';i++)
    {
       len++;
       j=0;
        
    }
    
    for(i=len-1;i>=0;i--)
    {
        str2[j]=str1[i];
        j++;
    }
    
    str2[j]='\0';
    
    cout<<"Reversed string:"<<str2;
}
