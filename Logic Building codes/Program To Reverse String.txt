#include <iostream>
#include<string.h>

using namespace std;


int main()
{
   int i,k;
   char c[25];
   cout<<"Enter a string: ";
   cin>>c;
   
   for(i=0;c[i]!='\0';i++)
   {
       k++;
   }
   
   cout<<"Reversed string is :";
   for(i=k-1;i>=0;i--)
   {
       cout<<c[i];
   }
}
