#include <iostream>

using namespace std;

int main()
{
    
    int i,j,n,temp;
    int arr[20];
    
    cout<<"Enter array number";
    cin>>n;
    cout<<"Enter the array elements";
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    
    for(i=0;i<n-1;i++) 
    {
        for(j=i+1;j<n;j++)
        {
            if(arr[i]>arr[j])
            {
                temp=arr[i];
                arr[i]=arr[j];
                arr[j]=temp;
            }
        }
    }
    
    cout<<"sorted array in ascending order :";
    for(i=0;i<n;i++)
    {
        cout<<"\n"<<arr[i];
    }
}
