#include <iostream>
#include<string.h>

using namespace std;


int main()
{
   int n,i,j,k=1,sp;
   
   cout<<"Enter the number: ";
   cin>>n;
   
   for(i=0;i<n;i++){
       
       for(sp=1;sp<=n-i;sp++){
           
           cout<<" ";
       }
       
       for(j=0;j<=i;j++){
           
           if(j==0 || i==0)
           {
               k=1;
           }
           else
           k=k*(i-j+1)/j;
           cout<<k<<"\t";
           
       }
       cout<<"\n";
   }
   
   
}
