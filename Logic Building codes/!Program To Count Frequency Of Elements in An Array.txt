#include <iostream>

using namespace std;

int main()
{
    int n,i,k,j;
    int arr[25],dup[25];
    cout<<"Enter the array number";
    cin>>n;
    cout<<"Enter the array elements:";
    for(i=0;i<n;i++){
        cin>>arr[i];
        dup[i]=0;
    }
    
    for(i=0;i<n;i++){
        
        k=1;
        
        for(j=i+1;j<n;j++){
            
            if(arr[i]==arr[j]){
                
                k++;
                dup[j]=1;
            }
            
        }
        
        if(dup[i]!=1){
            
            dup[i]=k;
        }
        
    }
    
    cout<<"Duplicate elements are:";
    for(i=0;i<n;i++){
        
        if(dup[i]!=1){
            
            cout<<"\n"<<arr[i]<<"\nTimes"<<dup[i];
        }
    }
    
    

    return 0;
}
