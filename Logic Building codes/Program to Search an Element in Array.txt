#include <iostream>

using namespace std;

int main()
{
    int n,i,k,flag=0;
    int arr[25];
    
    cout<<"Enter the array number";
    cin>>n;
    
    cout<<"Enter the array elements:";
    
    for(i=0;i<n;i++){
        
        cin>>arr[i];
        
    }
    
    cout<<"Enter the element to find:";
    cin>>k;
    
    for(i=0;i<n;i++)
    {
       
       if(arr[i]==k)
       {
           
           flag=1;
           cout<<"\nElement "<<k<<"found at postion "<<i+1;
           break;
           
       }
       
    }

   if(flag==0)
   {
       
       cout<<"Entered elemnt is not in the array!! "<<k;
   }
