#include <iostream>
#include<string.h>

using namespace std;


int main()
{
   int Xcnt=0;
   int Ycnt=0;
   int sum=3;
   
   do
   {
       if(Xcnt==Ycnt)
       {
           Xcnt++;
       }
       else
       {
           Ycnt++;
       }
       
       for(int i=0;i<Xcnt;i++)
       {
           cout<<"X";
       }
       for(int j=0;j<Ycnt;j++)
       {
           cout<<"Y";
       }
       
       cout<<"\n";
       
   }
   while(Ycnt<sum);
   
}
