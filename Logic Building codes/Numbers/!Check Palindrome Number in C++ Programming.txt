#include <iostream>

using namespace std;


int main()
{ 
  int n,rev=0,temp;
  cin>>n;

  temp=n;

  while(n!=0)
  {
    rev = rev * 10;
    rev = rev + (n % 10);
    n = n/10;

  }
  if(temp==rev){
    cout<< "1";
  }
  else
  {
    cout<<"0";
  }
}
